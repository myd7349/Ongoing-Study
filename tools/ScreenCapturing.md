    > gendef PrScrn.dll
    > 
    > i686-w64-mingw32-dlltool -v -d PrScrn.def -l PrScrn.lib
    > 
    > i686-w64-mingw32-g++ -Wall -O3 -static -s -o PrScrn.exe PrScrn.c -mwindows -nostdlib -luser32 -L. -lPrScrn
